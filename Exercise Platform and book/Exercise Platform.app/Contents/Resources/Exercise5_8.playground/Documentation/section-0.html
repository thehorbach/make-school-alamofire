5.8 Magic Time!

Use this magic * operator to solve the next challenge:

func *(string: String, scalar: Int) -> String {
    let array = Array(count: scalar, repeatedValue: string) 
    return array.joinWithSeparator("")
}

print("cat " * 3 + "dog " * 2)
// cat cat cat dog dog 

var newLine = "\n" * 2

print(newLine)
